  <script>
    function updateCountdown() {
      const countdownEl = document.getElementById('countdown');
      const eventTime = new Date('2025-05-25T00:00:00Z').getTime(); // UTC
      const now = new Date().getTime();
      const offset = 8 * 60 * 60 * 1000; // WITA UTC+8
      const diff = eventTime - now + offset;

      if (diff <= 0) {
        countdownEl.textContent = "Acara telah dimulai";
        return;
      }

      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
      const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
      const minutes = Math.floor((diff / (1000 * 60)) % 60);
      const seconds = Math.floor((diff / 1000) % 60);

      countdownEl.textContent = `${days} Hari ${hours} Jam ${minutes} Menit ${seconds} Detik`;
    }

    setInterval(updateCountdown, 1000);
    updateCountdown();
  </script>
